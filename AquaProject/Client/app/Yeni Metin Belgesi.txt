def LockScreen(self):#Bitti destek
        if not self.isAppOpen:
            self.isAppOpen = True
            subprocess.run("cd release && Aqua.exe",shell=True)
            

    def OpenScreen(self):#Bitti destek
        if self.isAppOpen:
            self.isAppOpen = False
            subprocess.run('taskkill /IM "Aqua.exe" /F', shell=True)
